# Assignement 4.2 : Problem Statement 2: Write a Python function which takes a character (i.e. a string of length 1) and returns True if it is
# a vowel, False otherwise.

def ChkVowel(v):
   
    if (v == 'a' or v == 'e' or v == 'i' or v == 'o' or v == 'u'):  
        return True
    else:
        return False
